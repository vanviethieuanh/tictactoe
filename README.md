### 🎨 How to add theme ?

```js
/* script.js */

const COLOR_THEMES = [
    {
        title: 'Strawberry',
        O: ['02AAB0', '00CDAC'],
        X: ['EB3349', 'F45C43'],
        lightTheme: true,
    },
    ...
    {
        title: 'Fahrenheit 451',  // Title of theme
        O: ['fc6076', 'ff9a44'],  // O graident stops
        X: ['fdfcfb', 'e2d1c3'],  // X gradient stops
        lightTheme: false,        // Is it light theme?
    },
]
```
